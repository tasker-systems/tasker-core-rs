def self.info
  loader = ConditionalLoader.instance
  base_info = {
    should_load: loader.should_load_test_environment?,
    loaded: loader.loaded,
    handler_count: loader.loaded_handler_count
  }

  if loader.loaded
    base_info.merge(loader.test_template_info)
  else
    base_info
  end
end
